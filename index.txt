    function updateColorField() {
      const type = document.getElementById('lockType').value;
      const color = document.getElementById('color');
      const greyOpt = color.querySelector('option[value="grey"]');
      if (type === 'battery-free') {
        color.disabled = false; greyOpt.disabled = false;
      } else if (type === 'code-key') {
        color.disabled = false; greyOpt.disabled = true;
        if (color.value === 'grey') color.value = '';
      } else {
        color.disabled = true; color.value = '';
      }
    }
